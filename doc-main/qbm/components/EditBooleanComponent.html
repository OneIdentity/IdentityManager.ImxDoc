import { Component } from '@angular/core';
import { UntypedFormControl } from '@angular/forms';

import { EditorBase } from '../editor-base';

/**
 * A component for viewing / editing boolean columns
 */
@Component({
  selector: 'imx-checkbox-editor',
  templateUrl: './edit-boolean.component.html',
  styleUrls: ['./edit-boolean.component.scss'],
})
export class EditBooleanComponent extends EditorBase<boolean> {
  public readonly control = new UntypedFormControl();
}

<mat-checkbox *ngIf="columnContainer?.canEdit; else isReadonly" [formControl]="control" [attr.data-imx-identifier]="'cdr-edit-boolean-checkbox-' + columnContainer?.name">
  <span>{{ columnContainer?.display | translate }}</span>
</mat-checkbox>
<mat-spinner class="imx-margin-left-10" diameter="16" *ngIf="isBusy"></mat-spinner>

<ng-template #isReadonly>
  <imx-view-property
    [columnContainer]="columnContainer"
    defaultValue="#LDS#Not set"
  >
  </imx-view-property>
</ng-template>
